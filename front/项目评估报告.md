#### 2. Token 存储不一致 (影响：高 | 优先级：P0)

**问题代码**:
```javascript
// api/index.js L12-18
let token = localStorage.getItem('front_token')
if (!token) {
  token = localStorage.getItem('token')  // 兼容旧 token
  if (token) {
    localStorage.setItem('front_token', token)
    localStorage.removeItem('token')
  }
}
```

**问题分析**:
- Token 迁移逻辑放在拦截器中不合适
- 应该在用户 store 初始化时统一处理
- 存在多个 token 名称（front_token, token, front_refresh）

**解决方案**:

```javascript
// stores/user.js
export const useUserStore = defineStore('user', () => {
  const token = ref('')
  const user = ref(null)
  
  // 初始化时处理 token 迁移
  const initToken = () => {
    // 优先使用新 token
    let savedToken = localStorage.getItem('front_token')
    
    // 如果没有，尝试迁移旧 token
    if (!savedToken) {
      const oldToken = localStorage.getItem('token')
      if (oldToken) {
        savedToken = oldToken
        localStorage.setItem('front_token', oldToken)
        localStorage.removeItem('token')
      }
    }
    
    // 迁移 refresh token
    const refreshToken = localStorage.getItem('front_refresh')
    if (!refreshToken) {
      const oldRefresh = localStorage.getItem('refresh')
      if (oldRefresh) {
        localStorage.setItem('front_refresh', oldRefresh)
        localStorage.removeItem('refresh')
      }
    }
    
    token.value = savedToken || ''
  }
  
  initToken()
  
  // ... 其他代码
})
```

#### 3. Profile 检查定时器内存泄漏风险 (影响：中高 | 优先级：P0)

**问题代码**:
```javascript
// stores/user.js L51-70
let profileCheckTimer = null

const startProfileCheck = () => {
  if (profileCheckTimer) {
    clearInterval(profileCheckTimer)
  }
  
  profileCheckTimer = setInterval(async () => {
    if (token.value) {
      try {
        const { data } = await api.get('/auth/profile/')
        // ...
      } catch (error) {
        console.error('Profile check error:', error)
      }
    }
  }, 30000)
}

const stopProfileCheck = () => {
  if (profileCheckTimer) {
    clearInterval(profileCheckTimer)
    profileCheckTimer = null
  }
}
```

**问题分析**:
- 页面销毁时没有自动清除定时器
- 依赖手动调用 `stopProfileCheck`，容易遗漏
- 多标签页场景会创建多个定时器
- 可能导致内存泄漏和重复请求

**解决方案**:

```javascript
// stores/user.js
import { watchEffect } from 'vue'

export const useUserStore = defineStore('user', () => {
  const token = ref(localStorage.getItem('front_token') || '')
  
  // 使用 watchEffect 自动管理定时器生命周期
  watchEffect((onCleanup) => {
    if (!token.value) {
      return
    }
    
    const checkProfile = async () => {
      try {
        const { data } = await api.get('/auth/profile/')
        if (JSON.stringify(user.value) !== JSON.stringify(data)) {
          user.value = data
          localStorage.setItem('front_user', JSON.stringify(data))
        }
      } catch (error) {
        console.error('Profile check error:', error)
      }
    }
    
    // 立即执行一次
    checkProfile()
    
    // 设置定时器
    const timerId = setInterval(checkProfile, 30000)
    
    // 清理函数会在 token 变化或 store 卸载时自动调用
    onCleanup(() => {
      clearInterval(timerId)
      console.log('Profile check cleaned up')
    })
  })
  
  // 保留手动控制方法（用于特殊情况）
  const forceStopProfileCheck = () => {
    token.value = ''  // 触发 watchEffect 清理
  }
  
  return {
    // ...
    forceStopProfileCheck
  }
})
```

```javascript
// composables/useArticleContent.js
import { ref, computed } from 'vue'
import { getContent } from '@/api/content'
import { marked } from 'marked'
import hljs from 'highlight.js'

export function useArticleContent() {
  const loading = ref(false)
  const article = ref({})
  const fullContentLoaded = ref(false)
  const headings = ref([])
  
  const renderedContent = computed(() => {
    if (!article.value.content) return ''
    
    marked.setOptions({
      highlight: function(code, lang) {
        if (lang && hljs.getLanguage(lang)) {
          try {
            return hljs.highlight(code, { language: lang }).value
          } catch (e) {
            console.error(e)
          }
        }
        return hljs.highlightAuto(code).value
      },
      breaks: true,
      gfm: true,
    })
    
    let html = marked.parse(article.value.content)
    extractHeadings(html)
    return html
  })
  
  const extractHeadings = (html) => {
    const tempDiv = document.createElement('div')
    tempDiv.innerHTML = html
    const headingElements = tempDiv.querySelectorAll('h1, h2, h3, h4, h5, h6')
    const result = []
    headingElements.forEach((el, index) => {
      const id = `heading-${index}`
      el.id = id
      result.push({ id, level: parseInt(el.tagName.charAt(1)), text: el.textContent })
    })
    headings.value = result
  }
  
  const loadArticle = async (id) => {
    loading.value = true
    try {
      const data = await getContent(id)
      article.value = data
      if (article.value.content === article.value.content_preview &&
          article.value.content.length === 5000 &&
          !fullContentLoaded.value) {
        loadFullContent(id)
      } else {
        fullContentLoaded.value = true
      }
    } catch (error) {
      console.error('Failed to load article:', error)
    } finally {
      loading.value = false
    }
  }
  
  const loadFullContent = async (id) => {
    try {
      const data = await getContent(id, { full: true })
      article.value = data
      fullContentLoaded.value = true
    } catch (error) {
      console.error('Failed to load full content:', error)
    }
  }
  
  return {
    loading,
    article,
    fullContentLoaded,
    headings,
    renderedContent,
    loadArticle,
    loadFullContent
  }
}

// composables/useArticleComments.js
import { ref } from 'vue'
import { getComments, createComment, likeComment } from '@/api/content'

export function useArticleComments(articleId) {

    }
  }
  
  const submitComment = async () => {

    }
  }

        content: replyContent.value,

      })

      replyToParent.value = null

    }
  }
  
  return {
    comments,
    commentContent,
    submitting,
    replyToParent,
```

```javascript
// composables/useErrorHandler.js
import { ElMessage } from 'element-plus'
import { ERROR_MESSAGES } from '@/constants/errorMessages'

export function useErrorHandler() {
  const handleError = (error, options = {}) => {
    const {
      showMessage = true,
      customMessage,
      ignoreStatus = []
    } = options
    
    console.error('Error:', error)
    
    // 忽略特定状态码
    if (ignoreStatus.includes(error.response?.status)) {
      return
    }
    
    // 自定义消息
    if (customMessage) {
      if (showMessage) {
        ElMessage.error(customMessage)
      }
      return
    }
    
    // 根据状态码显示消息
    const status = error.response?.status
    if (status === 401) {
      // 401 由拦截器统一处理
      return
    } else if (status === 403) {
      showMessage && ElMessage.error(ERROR_MESSAGES.FORBIDDEN)
    } else if (status === 404) {
      showMessage && ElMessage.error(ERROR_MESSAGES.NOT_FOUND)
    } else if (status === 500) {
      showMessage && ElMessage.error(ERROR_MESSAGES.SERVER_ERROR)
    } else if (error.code === 'ECONNABORTED') {
      showMessage && ElMessage.error(ERROR_MESSAGES.TIMEOUT)
    } else {
      showMessage && ElMessage.error(ERROR_MESSAGES.UNKNOWN)
    }
  }
  
  return {
    handleError
  }
}
```

#### user store

**缺点**:

```javascript
// stores/user.js
export const useUserStore = defineStore('user', () => {
  const token = ref('')
  const user = ref(null)
  const refreshToken = ref('')
  
  // 初始化
  const initFromStorage = () => {
    token.value = localStorage.getItem('front_token') || ''
    refreshToken.value = localStorage.getItem('front_refresh') || ''
    user.value = JSON.parse(localStorage.getItem('front_user') || 'null')
  }
  
  // 保存到 storage
  const saveToStorage = () => {
    if (token.value) {
      localStorage.setItem('front_token', token.value)
      localStorage.setItem('front_refresh', refreshToken.value)
      localStorage.setItem('front_user', JSON.stringify(user.value))
    } else {
      localStorage.removeItem('front_token')
      localStorage.removeItem('front_refresh')
      localStorage.removeItem('front_user')
    }
  }
  
  // 刷新 token
  const refreshAccessToken = async () => {
    if (!refreshToken.value) {
      throw new Error('No refresh token')
    }
    
    try {
      const { data } = await userApi.refreshToken({ refresh: refreshToken.value })
      token.value = data.access
      refreshToken.value = data.refresh
      saveToStorage()
    } catch (error) {
      // 刷新失败，清除所有 token
      token.value = ''
      refreshToken.value = ''
      user.value = null
      saveToStorage()
      throw error
    }
  }
  
  // 登录
  const login = async (credentials) => {
    const { data } = await userApi.login(credentials)
    token.value = data.access
    refreshToken.value = data.refresh
    user.value = data.user
    saveToStorage()
    return data
  }
  
  // 登出
  const logout = async () => {
    try {
      if (refreshToken.value) {
        await userApi.logout({ refresh: refreshToken.value })
      }
    } catch (e) {
      console.error(e)
    }
    
    token.value = ''
    refreshToken.value = ''
    user.value = null
    saveToStorage()
  }
  
  // 初始化
  initFromStorage()
  
  return {
    token,
    user,
    isLoggedIn: computed(() => !!token.value),
    login,
    logout,
    refreshAccessToken
  }
})
```

### 3. 敏感信息存储评分：6.0/10

**现状**:

- ❌ 没有加密存储
- ❌ localStorage 易受 XSS 攻击

**建议**:
- 考虑使用 httpOnly cookie（需要后端配合）
- 或者使用加密库存储敏感信息

```javascript
// 使用 crypto-js 加密存储
import CryptoJS from 'crypto-js'

const SECRET_KEY = import.meta.env.VITE_SECRET_KEY

const encryptToken = (token) => {
  return CryptoJS.AES.encrypt(token, SECRET_KEY).toString()
}

const decryptToken = (encryptedToken) => {
  const bytes = CryptoJS.AES.decrypt(encryptedToken, SECRET_KEY)
  return bytes.toString(CryptoJS.enc.Utf8)
}

// 存储
localStorage.setItem('encrypted_token', encryptToken(token))

// 读取
const token = decryptToken(localStorage.getItem('encrypted_token'))
```
